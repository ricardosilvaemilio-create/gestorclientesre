// =================================================================== // Script de ligação entre a Google Sheet "Base de dados de clientes" // e o painel de Gestão de Fidelizações. // // COMO USAR: // 1. Abre a tua Google Sheet (a que tem o separador "Geral" com os // dados dos clientes). // 2. Menu: Extensões > Apps Script. // 3. Apaga o conteúdo que lá estiver e cola este ficheiro inteiro. // 4. Clica em "Implementar" (Deploy) > "Nova implementação". // - Tipo: "Aplicação Web" (Web app) // - Executar como: Eu (a tua conta) // - Quem tem acesso: Qualquer pessoa (Anyone) // 5. Autoriza as permissões pedidas (é o teu próprio script, é seguro). // 6. Copia o URL gerado (termina em /exec) e envia-mo no chat. // // Mantém o TOKEN abaixo privado — é a única coisa que impede que // outra pessoa com o URL leia ou escreva na tua folha (protege // leituras e escritas). // =================================================================== var SHEET_NAME = 'Geral'; var TOKEN = 'vOJtAuuqkgE6_9Xo0hsOjjRm73-36jMi'; // não partilhes isto var HEADER_ROW = 1; var FIRST_DATA_ROW = 2; var LAST_COL = 19; // colunas A (Data Ativação) até S (Data do estado) // R = Estado do contacto, S = Data do estado — precisas de acrescentar estas // duas colunas (com cabeçalho) à folha antes de publicares esta versão. var PAGE_SIZE = 2000; // linhas por pedido — evita uma única execução gigante e lenta function doGet(e) { var token = e && e.parameter ? e.parameter.token : null; if (token !== TOKEN) { return jsonOut({ ok: false, error: 'token inválido' }); } var sheet = SpreadsheetApp.getActive().getSheetByName(SHEET_NAME); var lastRow = sheet.getLastRow(); var totalDataRows = Math.max(0, lastRow - FIRST_DATA_ROW + 1); var offset = parseInt((e.parameter && e.parameter.offset) || '0', 10) || 0; var limit = parseInt((e.parameter && e.parameter.limit) || String(PAGE_SIZE), 10) || PAGE_SIZE; if (totalDataRows <= 0) { return jsonOut({ headers: headerRow_(sheet), rows: [], total: 0, offset: 0, hasMore: false }); } var startRow = FIRST_DATA_ROW + offset; var numRows = Math.min(limit, lastRow - startRow + 1); if (numRows <= 0) { return jsonOut({ headers: headerRow_(sheet), rows: [], total: totalDataRows, offset: offset, hasMore: false }); } var values = sheet.getRange(startRow, 1, numRows, LAST_COL).getValues(); var out = []; for (var i = 0; i < values.length; i++) { var row = values[i]; if (row.join('') === '') continue; // ignora linhas totalmente vazias var formatted = row.map(formatCell); out.push([startRow + i].concat(formatted)); } var hasMore = (startRow + numRows - 1) < lastRow; return jsonOut({ headers: headerRow_(sheet), rows: out, total: totalDataRows, offset: offset, nextOffset: offset + numRows, hasMore: hasMore, generatedAt: new Date().toISOString() }); } function headerRow_(sheet) { return sheet.getRange(HEADER_ROW, 1, 1, LAST_COL).getValues()[0]; } function doPost(e) { try { var body = JSON.parse(e.postData.contents); if (body.token !== TOKEN) { return jsonOut({ ok: false, error: 'token inválido' }); } var sheet = SpreadsheetApp.getActive().getSheetByName(SHEET_NAME); var colLetters = ['A','B','C','D','E','F','G','H','I','J','K','L','M','N','O','P','Q','R','S']; if (body.action === 'create') { var newRow = sheet.getLastRow() + 1; var values = body.values || {}; for (var vKey in values) { var vIdx = colLetters.indexOf(vKey.toUpperCase()); if (vIdx > -1 && values[vKey] !== '' && values[vKey] != null) { sheet.getRange(newRow, vIdx + 1).setValue(values[vKey]); } } return jsonOut({ ok: true, row: newRow, created: true }); } if (body.action === 'sendProposal') { var to = body.to; var subject = body.subject || 'Proposta'; var htmlBody = body.html || ''; if (!to) { return jsonOut({ ok: false, error: 'sem e-mail do cliente' }); } MailApp.sendEmail({ to: to, subject: subject, htmlBody: htmlBody, body: 'A tua proposta segue neste e-mail — ativa a visualização em HTML para a veres formatada.' }); return jsonOut({ ok: true, sent: true }); } var row = body.row; if (!row || row < FIRST_DATA_ROW) { return jsonOut({ ok: false, error: 'linha inválida' }); } var lastRow = sheet.getLastRow(); if (row > lastRow) { return jsonOut({ ok: false, error: 'linha inválida' }); } if (body.action === 'delete') { sheet.deleteRow(row); return jsonOut({ ok: true, row: row, deleted: true }); } var updates = body.updates || {}; // ex: {"Q": "texto novo", "D": "2026-08-01"} var applied = {}; for (var key in updates) { var idx = colLetters.indexOf(key.toUpperCase()); if (idx > -1) { sheet.getRange(row, idx + 1).setValue(updates[key]); applied[key] = updates[key]; } } return jsonOut({ ok: true, row: row, applied: applied }); } catch (err) { return jsonOut({ ok: false, error: String(err) }); } } function formatCell(v) { // Nota: evitamos Utilities.formatDate aqui de propósito — é uma chamada de // serviço "pesada" e, repetida para milhares de células, torna o pedido // lentíssimo (chegou a demorar mais de um minuto com ~15.500 linhas). // Usamos apenas métodos nativos de Date, muito mais rápidos. if (Object.prototype.toString.call(v) === '[object Date]') { var y = v.getFullYear(); var m = ('0' + (v.getMonth() + 1)).slice(-2); var d = ('0' + v.getDate()).slice(-2); return y + '-' + m + '-' + d; } return v; } function jsonOut(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
